      # Gnuplot script file for plotting data in file "../data/Problem3_17.dat"
      # Plot with:
      # terminal```
      # >gnuplot
      # >load 'Problem3_17_3D.txt'
      # ```
      set parametric
      set style data points
      set   autoscale                        # scale axes automatically
      unset log                              # remove any log-scaling
      unset label                            # remove any previous labels
      set xtic auto                          # set xtics automatically
      set ytic auto                          # set ytics automatically
      set ztic auto
      set title "Problem 3.17 - Poincare Section"
      set xlabel "x [m]"
      set ylabel "y [m]"
      set zlabel "theta_{dot} [rad/s]"
      splot    "../data/Problem3_17.dat" using (cos($1)):(sin($1)):($2)
